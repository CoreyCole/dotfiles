```go
func (s *Service) BuildArtifactPane(
    ctx context.Context,
    threadID, runID, requestedRoot, selectedPath string,
) (ArtifactPaneState, error) {
    thread, err := s.queries.GetAgentThread(ctx, threadID)
    if err != nil {
        if errors.Is(err, sql.ErrNoRows) {
            return ArtifactPaneState{}, nil
        }
        return ArtifactPaneState{}, err
    }

    baseRoot := strings.TrimSpace(thread.Cwd)
    if baseRoot == "" {
        return ArtifactPaneState{}, nil
    }

    root := s.normalizeArtifactRoot(baseRoot, requestedRoot, selectedPath)
    files, err := listRenderableArtifacts(root)
    if err != nil {
        return ArtifactPaneState{}, err
    }

    selectedRel := resolveSelectedArtifactRelativePath(root, files, selectedPath)
    if selectedRel == "" {
        selectedRel = defaultArtifactPath(files)
    }

    selected, err := s.renderArtifact(root, selectedRel)
    if err != nil {
        return ArtifactPaneState{}, err
    }

    return ArtifactPaneState{
        ActiveRunID:  activeRunID,
        ArtifactRoot: root,
        RootLabel:    artifactRootLabel(root),
        WorkingDir:   s.buildWorkingDirectoryState(root, artifactResetPath(baseRoot, root)),
        Tree:         buildArtifactTree(files, selectedRel, root),
        Selected:     selected,
    }, nil
}
```

Add a root normalizer that never escapes the immutable thread cwd:

```go
func (s *Service) normalizeArtifactRoot(baseRoot, requestedRoot, selectedPath string) string {
    baseRoot = strings.TrimSpace(baseRoot)
    requestedRoot = strings.TrimSpace(requestedRoot)

    if requestedRoot != "" {
        if abs, err := filepath.Abs(requestedRoot); err == nil {
            requestedRoot = abs
        }
        if sameFilesystemPath(requestedRoot, baseRoot) || pathWithinRoot(requestedRoot, baseRoot) {
            return requestedRoot
        }
    }

    if focused := focusedArtifactRoot(baseRoot, selectedPath); focused != "" {
        if sameFilesystemPath(focused, baseRoot) || pathWithinRoot(focused, baseRoot) {
            return focused
        }
    }

    return baseRoot
}
```

#### 4. Add Go-side session JSONL helpers so terminal finalization can import a session delta

**`pkg/agents/conversation/session_jsonl.go`**

Write the base snapshot to a durable session file and later reconstruct the delta without relying on the TS worker:

```go
package conversation

import (
    "bufio"
    "encoding/json"
    "fmt"
    "os"
    "path/filepath"
    "sort"
    "strings"
    "time"
)

type sessionJSONLHeader struct {
    Type          string `json:"type"`
    Version       int    `json:"version"`
    ID            string `json:"id"`
    ParentSession string `json:"parentSession,omitempty"`
    Cwd           string `json:"cwd"`
}

type sessionJSONLEntry struct {
    Type      string `json:"type"`
    ID        string `json:"id"`
    ParentID  string `json:"parentId,omitempty"`
    Timestamp string `json:"timestamp"`
}

func WriteSessionJSONL(path string, snapshot Snapshot) error {
    if err := os.MkdirAll(filepath.Dir(path), 0o755); err != nil {
        return fmt.Errorf("mkdir session dir: %w", err)
    }

    header := sessionJSONLHeader{
        Type:    "session",
        Version: 3,
        ID:      strings.TrimSpace(snapshot.Header.SessionID),
        Cwd:     snapshot.Header.Cwd,
    }
    if header.ID == "" {
        header.ID = snapshot.LineageID
    }
    if parent := strings.TrimSpace(snapshot.Header.ParentSessionID); parent != "" {
        header.ParentSession = parent
    }

    file, err := os.Create(path)
    if err != nil {
        return fmt.Errorf("create session file: %w", err)
    }
    defer file.Close()

    enc := json.NewEncoder(file)
    if err := enc.Encode(header); err != nil {
        return fmt.Errorf("encode session header: %w", err)
    }
    for _, entry := range snapshot.Entries {
        if _, err := file.WriteString(strings.TrimSpace(entry.PayloadJSON) + "\n"); err != nil {
            return fmt.Errorf("write snapshot entry: %w", err)
        }
    }
    return nil
}

func BuildSessionDelta(sessionPath string, base Snapshot) (Checkpoint, error) {
    file, err := os.Open(sessionPath)
    if err != nil {
        return Checkpoint{}, fmt.Errorf("open session file: %w", err)
    }
    defer file.Close()

    existing := make(map[string]struct{}, len(base.Entries))
    nextOriginOrder := int64(0)
    for _, entry := range base.Entries {
        existing[entry.EntryID] = struct{}{}
        if entry.OriginOrder >= nextOriginOrder {
            nextOriginOrder = entry.OriginOrder + 1
        }
    }

    scanner := bufio.NewScanner(file)
    scanner.Buffer(make([]byte, 0, 64*1024), 8*1024*1024)
    if !scanner.Scan() {
        return Checkpoint{}, fmt.Errorf("session file missing header")
    }

    var header sessionJSONLHeader
    if err := json.Unmarshal(scanner.Bytes(), &header); err != nil {
        return Checkpoint{}, fmt.Errorf("decode session header: %w", err)
    }

    entries := make([]SnapshotEntry, 0)
    leafID := base.HeadEntryID
    for scanner.Scan() {
        raw := append([]byte(nil), scanner.Bytes()...)
        var probe sessionJSONLEntry
        if err := json.Unmarshal(raw, &probe); err != nil {
            return Checkpoint{}, fmt.Errorf("decode session entry: %w", err)
        }
        if strings.TrimSpace(probe.ID) == "" {
            continue
        }
        leafID = probe.ID
        if _, ok := existing[probe.ID]; ok {
            continue
        }
        ts, err := time.Parse(time.RFC3339Nano, probe.Timestamp)
        if err != nil {
            return Checkpoint{}, fmt.Errorf("parse session timestamp: %w", err)
        }
        entries = append(entries, SnapshotEntry{
            LineageID:     base.LineageID,
            EntryID:       probe.ID,
            ParentEntryID: probe.ParentID,
            EntryType:     probe.Type,
            Timestamp:     ts,
            OriginOrder:   nextOriginOrder,
            PayloadJSON:   string(raw),
        })
        nextOriginOrder++
    }
    if err := scanner.Err(); err != nil {
        return Checkpoint{}, fmt.Errorf("scan session file: %w", err)
    }

    sort.SliceStable(entries, func(i, j int) bool { return entries[i].OriginOrder < entries[j].OriginOrder })

    return Checkpoint{
        HeadEntryID: leafID,
        Header: SnapshotHeader{
            SessionID:       header.ID,
            ParentSessionID: header.ParentSession,
            Cwd:             header.Cwd,
        },
        NewEntries: entries,
    }, nil
}
```

```go
package conversationworkflow

import (
    "fmt"
    "time"

    conversation "github.com/premiumlabs/agents/pkg/agents/conversation"
    temporalmgr "github.com/premiumlabs/agents/pkg/agents/temporal"
    "go.temporal.io/sdk/workflow"
)

func RunTerminalSessionWorkflow(ctx workflow.Context, input conversation.TerminalWorkflowInput) (conversation.RunResult, error) {
    state := "starting"
    _ = workflow.SetQueryHandler(ctx, "state", func() (string, error) { return state, nil })

    actCtx := workflow.WithActivityOptions(ctx, workflow.ActivityOptions{
        TaskQueue:           temporalmgr.GoTaskQueue,
        StartToCloseTimeout: 30 * time.Second,
    })

    attachCh := workflow.GetSignalChannel(ctx, string(temporalmgr.SignalTerminalAttach))
    detachCh := workflow.GetSignalChannel(ctx, string(temporalmgr.SignalTerminalDetach))
    stopCh := workflow.GetSignalChannel(ctx, string(temporalmgr.SignalTerminalStop))
    exitCh := workflow.GetSignalChannel(ctx, string(temporalmgr.SignalTerminalExit))
    finalizedCh := workflow.GetSignalChannel(ctx, string(temporalmgr.SignalTerminalFinalized))

    startupTimer := workflow.NewTimer(ctx, time.Duration(input.StartupTimeoutSeconds)*time.Second)

    waitForFinalized := func(reason string) error {
        state = reason
        finalizeTimer := workflow.NewTimer(ctx, time.Duration(input.FinalizeTimeoutSeconds)*time.Second)
        for {
            selector := workflow.NewSelector(ctx)
            selector.AddReceive(finalizedCh, func(ch workflow.ReceiveChannel, more bool) {
                var sig conversation.TerminalFinalizedSignal
                ch.Receive(ctx, &sig)
                state = "finalized_" + sig.Status
            })
            selector.AddFuture(finalizeTimer, func(workflow.Future) {
                state = "finalize_timeout"
                _ = workflow.ExecuteActivity(actCtx, "FailTerminalRunIfStillRunning", conversation.TerminalFailActivityInput{
                    RunID:        input.RunID,
                    SessionPath:  input.SessionPath,
                    ErrorMessage: "terminal finalization timed out",
                }).Get(ctx, nil)
            })
            selector.Select(ctx)
            if state == "finalized_complete" || state == "finalized_failed" || state == "finalize_timeout" {
                return nil
            }
        }
    }

    for {
        selector := workflow.NewSelector(ctx)
        selector.AddReceive(attachCh, func(ch workflow.ReceiveChannel, more bool) {
            var sig conversation.TerminalAttachSignal
            ch.Receive(ctx, &sig)
            state = "attached"
            startupTimer = nil
        })
        selector.AddReceive(detachCh, func(ch workflow.ReceiveChannel, more bool) {
            var sig conversation.TerminalDetachSignal
            ch.Receive(ctx, &sig)
            state = "detached"
        })
        selector.AddReceive(stopCh, func(ch workflow.ReceiveChannel, more bool) {
            var sig conversation.TerminalStopSignal
            ch.Receive(ctx, &sig)
            state = "stopping"
            _ = workflow.ExecuteActivity(actCtx, "StopTerminalRunIfStillRunning", conversation.TerminalStopActivityInput{
                RunID:  input.RunID,
                Reason: sig.Reason,
            }).Get(ctx, nil)
        })
        selector.AddReceive(exitCh, func(ch workflow.ReceiveChannel, more bool) {
            var sig conversation.TerminalExitSignal
            ch.Receive(ctx, &sig)
            state = "exited"
        })
        selector.AddReceive(finalizedCh, func(ch workflow.ReceiveChannel, more bool) {
            var sig conversation.TerminalFinalizedSignal
            ch.Receive(ctx, &sig)
            state = "finalized_" + sig.Status
        })
        if startupTimer != nil {
            selector.AddFuture(startupTimer, func(workflow.Future) {
                state = "startup_timeout"
                _ = workflow.ExecuteActivity(actCtx, "StopTerminalRunIfStillRunning", conversation.TerminalStopActivityInput{
                    RunID:  input.RunID,
                    Reason: "terminal startup timed out",
                }).Get(ctx, nil)
            })
        }
        selector.Select(ctx)

        switch state {
        case "finalized_complete", "finalized_failed":
            return conversation.RunResult{
                RunID:        input.RunID,
                ThreadID:     input.ThreadID,
                SessionPath:  input.SessionPath,
                ArtifactRoot: input.ArtifactRoot,
                MetadataJSON: fmt.Sprintf(`{"terminal_workflow_state":%q}`, state),
            }, nil
        case "startup_timeout", "stopping", "exited":
            if err := waitForFinalized(state); err != nil {
                return conversation.RunResult{}, err
            }
            return conversation.RunResult{
                RunID:        input.RunID,
                ThreadID:     input.ThreadID,
                SessionPath:  input.SessionPath,
                ArtifactRoot: input.ArtifactRoot,
            }, nil
        case "detached":
            reconnectTimer := workflow.NewTimer(ctx, time.Duration(input.ReconnectGraceSeconds)*time.Second)
            waiting := true
            for waiting {
                inner := workflow.NewSelector(ctx)
                inner.AddReceive(attachCh, func(ch workflow.ReceiveChannel, more bool) {
                    var sig conversation.TerminalAttachSignal
                    ch.Receive(ctx, &sig)
                    state = "attached"
                    waiting = false
                })
                inner.AddReceive(stopCh, func(ch workflow.ReceiveChannel, more bool) {
                    var sig conversation.TerminalStopSignal
                    ch.Receive(ctx, &sig)
                    state = "stopping"
                    waiting = false
                    _ = workflow.ExecuteActivity(actCtx, "StopTerminalRunIfStillRunning", conversation.TerminalStopActivityInput{RunID: input.RunID, Reason: sig.Reason}).Get(ctx, nil)
                })
                inner.AddReceive(exitCh, func(ch workflow.ReceiveChannel, more bool) {
                    var sig conversation.TerminalExitSignal
                    ch.Receive(ctx, &sig)
                    state = "exited"
                    waiting = false
                })
                inner.AddReceive(finalizedCh, func(ch workflow.ReceiveChannel, more bool) {
                    var sig conversation.TerminalFinalizedSignal
                    ch.Receive(ctx, &sig)
                    state = "finalized_" + sig.Status
                    waiting = false
                })
                inner.AddFuture(reconnectTimer, func(workflow.Future) {
                    state = "reconnect_timeout"
                    waiting = false
                    _ = workflow.ExecuteActivity(actCtx, "StopTerminalRunIfStillRunning", conversation.TerminalStopActivityInput{
                        RunID:  input.RunID,
                        Reason: "terminal reconnect grace expired",
                    }).Get(ctx, nil)
                })
                inner.Select(ctx)
            }
            if state == "reconnect_timeout" {
                if err := waitForFinalized(state); err != nil {
                    return conversation.RunResult{}, err
                }
                return conversation.RunResult{RunID: input.RunID, ThreadID: input.ThreadID, SessionPath: input.SessionPath, ArtifactRoot: input.ArtifactRoot}, nil
            }
        }
    }
}
```

```go
func (s *Service) finalizeTerminalRunOnce(ctx context.Context, input conversation.TerminalFinalizeInput) error {
    baseSnapshot, err := s.BuildSnapshot(ctx, input.ThreadID, input.ExpectedHeadID)
    if err != nil {
        return err
    }
    delta, err := conversation.BuildSessionDelta(input.SessionPath, baseSnapshot)
    if err != nil {
        return s.FailTerminalRunIfStillRunning(ctx, input.RunID, input.SessionPath, input.MetadataJSON, err.Error())
    }

    tx, err := s.db.BeginTx(ctx, nil)
    if err != nil {
        return err
    }
    defer tx.Rollback()

    q := s.queries.WithTx(tx)
    run, err := q.GetAgentRun(ctx, input.RunID)
    if err != nil {
        return err
    }
    if run.Status != "running" {
        return nil
    }

    nextHead := input.ExpectedHeadID
    if delta.HeadEntryID != "" {
        nextHead = delta.HeadEntryID
    }

    rows, err := q.UpdateAgentThreadHeadIfMatch(ctx, db.UpdateAgentThreadHeadIfMatchParams{
        ID:                  input.ThreadID,
        ExpectedHeadEntryID: sql.NullString{String: input.ExpectedHeadID, Valid: input.ExpectedHeadID != ""},
        NextHeadEntryID:     sql.NullString{String: nextHead, Valid: nextHead != ""},
    })
    if err != nil {
        return err
    }
    if rows == 0 {
        staleMeta := fmt.Sprintf(`{"reason":"stale_base_head","session_path":%q,"artifact_root":%q,"expected_head_entry_id":%q}`,
            input.SessionPath, input.ArtifactRoot, input.ExpectedHeadID,
        )
        _, err := q.FailAgentRun(ctx, db.FailAgentRunParams{
            ID:           input.RunID,
            SessionPath:  sql.NullString{String: input.SessionPath, Valid: true},
            MetadataJson: sql.NullString{String: staleMeta, Valid: true},
            ErrorMessage: sql.NullString{String: "terminal finalization base head no longer matches thread", Valid: true},
        })
        if err != nil {
            return err
        }
        if err := tx.Commit(); err != nil {
            return err
        }
        s.deleteTerminalSession(input.RunID)
        s.notifier.Notify(input.ThreadID, ThreadStreamSignal{Scope: PatchMainPane})
        _ = s.signalTerminalWorkflow(input.RunID, temporalmgr.SignalTerminalFinalized, conversation.TerminalFinalizedSignal{Status: "failed"})
        return nil
    }

    for _, entry := range delta.NewEntries {
        if err := q.CreateAgentEntry(ctx, db.CreateAgentEntryParams{
            LineageID:        baseSnapshot.LineageID,
            EntryID:          entry.EntryID,
            ParentEntryID:    sql.NullString{String: entry.ParentEntryID, Valid: entry.ParentEntryID != ""},
            EntryType:        entry.EntryType,
            OriginOrder:      entry.OriginOrder,
            PayloadJson:      entry.PayloadJSON,
            OriginThreadID:   input.ThreadID,
            OriginRunID:      sql.NullString{String: input.RunID, Valid: true},
            SessionTimestamp: entry.Timestamp,
        }); err != nil {
            return err
        }
    }

    metadataJSON := input.MetadataJSON
    if metadataJSON == "" {
        metadataJSON = fmt.Sprintf(`{"reason":"terminal_finalized","session_path":%q}`, input.SessionPath)
    }

    if input.ErrorMessage != "" {
        _, err = q.FailAgentRun(ctx, db.FailAgentRunParams{
            ID:           input.RunID,
            SessionPath:  sql.NullString{String: input.SessionPath, Valid: true},
            MetadataJson: sql.NullString{String: metadataJSON, Valid: metadataJSON != ""},
            ErrorMessage: sql.NullString{String: input.ErrorMessage, Valid: true},
        })
    } else {
        _, err = q.CompleteAgentRun(ctx, db.CompleteAgentRunParams{
            ID:                input.RunID,
            ResultHeadEntryID: sql.NullString{String: nextHead, Valid: nextHead != ""},
            SessionPath:       sql.NullString{String: input.SessionPath, Valid: true},
            MetadataJson:      sql.NullString{String: metadataJSON, Valid: metadataJSON != ""},
        })
    }
    if err != nil {
        return err
    }

    if err := tx.Commit(); err != nil {
        return err
    }

    s.resetLiveThread(input.ThreadID)
    s.deleteTerminalSession(input.RunID)
    s.notifier.NotifyScopes(input.ThreadID, PatchMainPane, PatchStableTranscript, PatchArtifactPane, PatchSidebar)

    if input.ErrorMessage != "" {
        status = "failed"
    }
    _ = s.signalTerminalWorkflow(input.RunID, temporalmgr.SignalTerminalFinalized, conversation.TerminalFinalizedSignal{Status: status})
    return nil
}
```

#### 2. Add the attach logic and PTY/websocket bridge

**`server/services/agentchat/terminal.go`**

Bridge one browser websocket to one PTY. Use JSON text frames for client control/input and binary frames for PTY output.

```go
type terminalClientMessage struct {
    Type string `json:"type"`
    Data string `json:"data,omitempty"`
    Cols uint16 `json:"cols,omitempty"`
    Rows uint16 `json:"rows,omitempty"`
}

func (s *Service) AttachTerminal(ctx context.Context, userEmail, threadID, runID string, conn TerminalConn) error {
    thread, err := s.queries.GetAgentThreadForUser(ctx, db.GetAgentThreadForUserParams{ID: threadID, UserEmail: userEmail})
    if err != nil {
        return err
    }
    run, err := s.queries.GetAgentRun(ctx, runID)
    if err != nil {
        return err
    }
    if run.ThreadID != thread.ID || run.Mode != string(conversation.RunModeTerminal) || run.Status != "running" {
        return fmt.Errorf("terminal run is not attachable")
    }

    session := s.getTerminalSession(run.ID)
    if session == nil {
        return fmt.Errorf("terminal session is not available in this process")
    }

    connID := uuid.NewString()
    if !s.claimTerminalAttachment(session, connID) {
        return fmt.Errorf("terminal is already attached")
    }
    defer s.releaseTerminalAttachment(session, connID)

    _ = s.signalTerminalWorkflow(run.ID, temporalmgr.SignalTerminalAttach, conversation.TerminalAttachSignal{})

    attachCtx, cancel := context.WithCancel(ctx)
    defer cancel()

    go func() {
        buf := make([]byte, 4096)
        for {
            n, err := session.PTY.Read(buf)
            if n > 0 {
                if writeErr := conn.WriteMessage(websocket.BinaryMessage, append([]byte(nil), buf[:n]...)); writeErr != nil {
                    cancel()
                    return
                }
            }
            if err != nil {
                cancel()
                return
            }
        }
    }()

    for {
        select {
        case <-attachCtx.Done():
            unexpected := runStillRunning(session.Cmd)
            if unexpected {
                _ = s.signalTerminalWorkflow(run.ID, temporalmgr.SignalTerminalDetach, conversation.TerminalDetachSignal{})
                s.startDetachTimer(session)
            }
            _ = conn.Close()
            return nil
        default:
            messageType, payload, err := conn.ReadMessage()
            if err != nil {
                cancel()
                continue
            }
            if messageType != websocket.TextMessage {
                continue
            }

            var msg terminalClientMessage
            if err := json.Unmarshal(payload, &msg); err != nil {
                continue
            }
            switch msg.Type {
            case "input":
                if _, err := io.WriteString(session.PTY, msg.Data); err != nil {
                    cancel()
                }
            case "resize":
                _ = pty.Setsize(session.PTY, &pty.Winsize{Cols: msg.Cols, Rows: msg.Rows})
            }
        }
    }
}
```

`startDetachTimer` should call `StopTerminalRunIfStillRunning` after the configured reconnect grace if the same session is still detached.

**`static/js/agent-chat-terminal.js`**

Use a plain ESM file, import xterm from CDN at runtime, and keep the browser logic isolated from Datastar app state.

```js
const XTERM_CSS_URL = 'https://cdn.jsdelivr.net/npm/@xterm/xterm@5.5.0/css/xterm.min.css';
const XTERM_MODULE_URL = 'https://cdn.jsdelivr.net/npm/@xterm/xterm@5.5.0/+esm';
const FIT_MODULE_URL = 'https://cdn.jsdelivr.net/npm/@xterm/addon-fit@0.10.0/+esm';

function ensureStylesheet() {
  if (document.querySelector(`link[data-agent-chat-terminal-css="${XTERM_CSS_URL}"]`)) {
    return;
  }
  const link = document.createElement('link');
  link.rel = 'stylesheet';
  link.href = XTERM_CSS_URL;
  link.dataset.agentChatTerminalCss = XTERM_CSS_URL;
  document.head.appendChild(link);
}

const instances = new Map();

async function loadTerminalLibs() {
  const [{ Terminal }, { FitAddon }] = await Promise.all([
    import(XTERM_MODULE_URL),
    import(FIT_MODULE_URL),
  ]);
  return { Terminal, FitAddon };
}

async function mount(el) {
  if (!el || !el.dataset.connectUrl || instances.has(el)) {
    return;
  }

  ensureStylesheet();
  const { Terminal, FitAddon } = await loadTerminalLibs();
  const term = new Terminal({
    convertEol: true,
    cursorBlink: true,
    theme: {
      background: '#000000',
      foreground: '#f8fafc',
    },
  });
  const fitAddon = new FitAddon();
  term.loadAddon(fitAddon);
  term.open(el);
  fitAddon.fit();

  const url = new URL(el.dataset.connectUrl, window.location.origin);
  url.protocol = url.protocol === 'https:' ? 'wss:' : 'ws:';
  const socket = new WebSocket(url);
  socket.binaryType = 'arraybuffer';

  const sendResize = () => {
    fitAddon.fit();
    socket.send(JSON.stringify({
      type: 'resize',
      cols: term.cols,
      rows: term.rows,
    }));
  };

  term.onData((data) => {
    if (socket.readyState === WebSocket.OPEN) {
      socket.send(JSON.stringify({ type: 'input', data }));
    }
  });

  socket.addEventListener('open', () => {
    term.focus();
    sendResize();
  });

  socket.addEventListener('message', (event) => {
    if (typeof event.data === 'string') {
      try {
        const payload = JSON.parse(event.data);
        if (payload.type === 'status' && payload.message) {
          term.writeln(`\r\n${payload.message}`);
        }
      } catch {
        term.write(event.data);
      }
      return;
    }
    const bytes = event.data instanceof ArrayBuffer ? new Uint8Array(event.data) : event.data;
    term.write(bytes);
  });

  socket.addEventListener('close', () => {
    term.writeln('\r\n[terminal disconnected]');
  });

  const resizeObserver = new ResizeObserver(() => {
    if (socket.readyState === WebSocket.OPEN) {
      sendResize();
    }
  });
  resizeObserver.observe(el);

  const destroy = () => {
    resizeObserver.disconnect();
    socket.close();
    term.dispose();
    instances.delete(el);
  };

  instances.set(el, destroy);
}

window.AgentChatTerminal = { mount };
```

No frontend transcript state is stored here. It is a pure terminal transport/view layer.

#### 1. Make detach/reconnect and stop/exit converge on one idempotent finalizer

**`server/services/agentchat/terminal.go`**

Tighten the session registry helpers so a detached websocket, explicit stop, PTY exit, or workflow timeout all flow through the same guarded code.

Key rules to implement:
- only one `AttachedConnID` at a time
- `startDetachTimer(session)` only starts when the terminal unexpectedly loses its controlling connection
- reconnect cancels the existing timer
- `FinalizedOnce` protects the in-process path, but DB `status = 'running'` guards remain authoritative

Representative helpers:

```go
func (s *Service) claimTerminalAttachment(session *terminalSession, connID string) bool {
    s.terminalMu.Lock()
    defer s.terminalMu.Unlock()
    if session.AttachedConnID != "" {
        return false
    }
    if session.DetachTimer != nil {
        session.DetachTimer.Stop()
        session.DetachTimer = nil
    }
    session.AttachedConnID = connID
    return true
}

func (s *Service) releaseTerminalAttachment(session *terminalSession, connID string) {
    s.terminalMu.Lock()
    defer s.terminalMu.Unlock()
    if session.AttachedConnID == connID {
        session.AttachedConnID = ""
    }
}

func (s *Service) startDetachTimer(session *terminalSession) {
    s.terminalMu.Lock()
    defer s.terminalMu.Unlock()
    if session.DetachTimer != nil {
        session.DetachTimer.Stop()
    }
    session.DetachTimer = time.AfterFunc(s.terminalConfig.ReconnectGrace, func() {
        _ = s.StopTerminalRunIfStillRunning(context.Background(), conversation.TerminalStopActivityInput{
            RunID:  session.RunID,
            Reason: "terminal reconnect grace expired",
        })
    })
}
```
